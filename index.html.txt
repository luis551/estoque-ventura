    <!-- Script Principal (O C√≥digo React adaptado) -->
    <script type="text/babel" data-type="module">
        import React, { useState, useEffect } from 'react';
        import { createRoot } from 'react-dom/client';
        import { 
            Plus, Trash2, FileSpreadsheet, Sparkles, Settings, X, 
            Database, RefreshCw, CloudDownload, Save 
        } from 'lucide-react';
        import { initializeApp } from 'firebase/app';
        import { 
            getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged 
        } from 'firebase/auth';
        import { 
            getFirestore, collection, doc, setDoc, deleteDoc, onSnapshot, writeBatch 
        } from 'firebase/firestore';

        // --- Utils e Configs ---

        const useSheetJS = () => {
            const [loaded, setLoaded] = useState(false);
            useEffect(() => {
                if (window.XLSX) { setLoaded(true); return; }
                const script = document.createElement('script');
                script.src = "https://cdn.sheetjs.com/xlsx-latest/package/dist/xlsx.full.min.js";
                script.async = true;
                script.onload = () => setLoaded(true);
                document.body.appendChild(script);
            }, []);
            return loaded;
        };

        const defaultData = [
            { id: '1', product: 'HEINEKEN', category: 'CERVEJAS', initial: 0, entry: 0, sales: 0, internal: 0, voucher: 0, damage: 0, physicalCount: 0 },
            { id: '2', product: 'IMPERIO', category: 'CERVEJAS', initial: 0, entry: 0, sales: 0, internal: 0, voucher: 0, damage: 0, physicalCount: 0 },
            { id: '11', product: 'DUNHILL', category: 'TABACARIA', initial: 0, entry: 10, sales: 5, internal: 0, voucher: 0, damage: 0, physicalCount: 5 },
            { id: '21', product: 'CACHA√áA 51', category: 'DESTILADOS', initial: 5, entry: 0, sales: 1, internal: 0, voucher: 0, damage: 0, physicalCount: 4 },
        ];

        function StockSheet() {
            // --- Estados ---
            const [data, setData] = useState([]);
            const [loading, setLoading] = useState(true);
            const [user, setUser] = useState(null);
            const [useLocalStorage, setUseLocalStorage] = useState(true);
            const [dbInstance, setDbInstance] = useState(null);
            
            // UI States
            const [apiKey, setApiKey] = useState(localStorage.getItem('gemini_api_key') || '');
            const [showConfig, setShowConfig] = useState(false);
            const [showAIModal, setShowAIModal] = useState(false);
            const [aiLoading, setAiLoading] = useState(false);
            const [aiResult, setAiResult] = useState('');
            
            // Modals
            const [showAddModal, setShowAddModal] = useState(false);
            const [newProductName, setNewProductName] = useState('');
            const [newCategory, setNewCategory] = useState('');
            const [customCategory, setCustomCategory] = useState('');
            const [showImportModal, setShowImportModal] = useState(false);
            const [importUrl, setImportUrl] = useState(localStorage.getItem('last_sheet_url') || '');
            const [importLoading, setImportLoading] = useState(false);

            const appId = 'estoque-ventura-web'; 

            useSheetJS();

            // --- Inicializa√ß√£o (Firebase ou Local) ---
            useEffect(() => {
                const initApp = async () => {
                    // Tenta pegar config do Firebase se o mestre injetou (no ambiente real, tu substituiria o null abaixo)
                    // Para facilitar, deixei preparado pra LocalStorage, mas se tu configurar, ele conecta.
                    const firebaseConfigRaw = null; 
                    const firebaseConfig = (typeof window.__firebase_config !== 'undefined') ? JSON.parse(window.__firebase_config) : firebaseConfigRaw;

                    if (!firebaseConfig) {
                        console.log("üíæ Modo LocalStorage Ativado");
                        const saved = localStorage.getItem('estoque_ventura_data');
                        setData(saved ? JSON.parse(saved) : defaultData);
                        setLoading(false);
                        return;
                    }

                    try {
                        const app = initializeApp(firebaseConfig);
                        const auth = getAuth(app);
                        const db = getFirestore(app);
                        setDbInstance(db);
                        setUseLocalStorage(false);

                        await signInAnonymously(auth);

                        onAuthStateChanged(auth, (u) => {
                            setUser(u);
                            if (!u) setLoading(false);
                        });
                    } catch (e) {
                        console.error("Erro Firebase:", e);
                        setUseLocalStorage(true);
                        const saved = localStorage.getItem('estoque_ventura_data');
                        setData(saved ? JSON.parse(saved) : defaultData);
                        setLoading(false);
                    }
                };
                initApp();
            }, []);

            // --- Listener de Dados ---
            useEffect(() => {
                if (useLocalStorage || !user || !dbInstance) return;
                const inventoryRef = collection(dbInstance, 'artifacts', appId, 'users', user.uid, 'inventory');
                
                const unsubscribe = onSnapshot(inventoryRef, (snapshot) => {
                    if (snapshot.empty && loading) {
                        const batch = writeBatch(dbInstance);
                        defaultData.forEach(item => {
                            const docRef = doc(inventoryRef, item.id);
                            const { id, ...rest } = item;
                            batch.set(docRef, { ...rest, updatedAt: new Date().toISOString() });
                        });
                        batch.commit();
                        setData(defaultData);
                    } else {
                        const items = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                        setData(items);
                    }
                    setLoading(false);
                });
                return () => unsubscribe();
            }, [user, dbInstance, useLocalStorage]);

            useEffect(() => { if (apiKey) localStorage.setItem('gemini_api_key', apiKey); }, [apiKey]);

            const saveLocal = (newData) => {
                setData(newData);
                localStorage.setItem('estoque_ventura_data', JSON.stringify(newData));
            };

            // --- A√ß√µes ---
            const handleUpdate = async (id, field, value) => {
                const newData = data.map(item => item.id === id ? { ...item, [field]: value } : item);
                setData(newData);
                if (useLocalStorage) saveLocal(newData);
                else if (user && dbInstance) {
                    const item = newData.find(i => i.id === id);
                    const { id: _, ...docData } = item;
                    await setDoc(doc(dbInstance, 'artifacts', appId, 'users', user.uid, 'inventory', id), { ...docData, updatedAt: new Date().toISOString() }, { merge: true });
                }
            };

            const handleAddRow = async () => {
                const newId = String(Date.now());
                const cat = customCategory.trim() ? customCategory.toUpperCase() : (newCategory || 'GERAL');
                const newItem = {
                    id: newId, product: newProductName.trim().toUpperCase(), category: cat,
                    initial: 0, entry: 0, sales: 0, internal: 0, voucher: 0, damage: 0, physicalCount: 0, updatedAt: new Date().toISOString()
                };
                if (useLocalStorage) saveLocal([...data, newItem]);
                else if (user && dbInstance) {
                    const { id, ...docData } = newItem;
                    await setDoc(doc(dbInstance, 'artifacts', appId, 'users', user.uid, 'inventory', newId), docData);
                }
                setShowAddModal(false); setNewProductName(''); setNewCategory(''); setCustomCategory('');
            };

            const handleDelete = async (id) => {
                if (!confirm("Certeza que quer apagar esse item, mestre?")) return;
                if (useLocalStorage) saveLocal(data.filter(i => i.id !== id));
                else if (user && dbInstance) await deleteDoc(doc(dbInstance, 'artifacts', appId, 'users', user.uid, 'inventory', id));
            };

            const handleExport = () => {
                if (!window.XLSX) return alert("Calma mestre, a magia do Excel t√° carregando...");
                const exportData = data.map(row => {
                    const final = (Number(row.initial)||0)+(Number(row.entry)||0)-(Number(row.sales)||0)-(Number(row.internal)||0)-(Number(row.voucher)||0)-(Number(row.damage)||0);
                    return {
                        "PRODUTO": row.product, "CATEGORIA": row.category, "INICIAL": row.initial, "ENTRADA": row.entry,
                        "VENDAS": row.sales, "CONSUMO": row.internal, "VALE": row.voucher, "AVARIA": row.damage,
                        "FINAL (SIST)": final, "CONTAGEM": row.physicalCount, "DIFEREN√áA": final - (Number(row.physicalCount)||0)
                    };
                });
                const wb = window.XLSX.utils.book_new();
                window.XLSX.utils.book_append_sheet(wb, window.XLSX.utils.json_to_sheet(exportData), "Estoque");
                window.XLSX.writeFile(wb, `ESTOQUE_VENTURA_${new Date().toLocaleDateString().replace(/\//g, '-')}.xlsx`);
            };

            const handleAI = async () => {
                if (!apiKey) { setShowConfig(true); return alert("Cad√™ a chave do Gemini, mestre?"); }
                setAiLoading(true); setShowAIModal(true); setAiResult('');
                const report = data.map(r => {
                    const f = (Number(r.initial)||0)+(Number(r.entry)||0)-(Number(r.sales)||0)-(Number(r.internal)||0)-(Number(r.voucher)||0)-(Number(r.damage)||0);
                    const d = f - (Number(r.physicalCount)||0);
                    return d !== 0 ? `${r.product}: Sist(${f}) vs Real(${r.physicalCount}) = Dif(${d})` : null;
                }).filter(Boolean);
                
                try {
                    const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST', headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: `Analise essas sobras/faltas de estoque de um bar carioca. Seja engra√ßado e direto. Use emojis. Dados: ${report.join('\n') || 'Tudo perfeito!'}` }] }] })
                    });
                    const json = await res.json();
                    setAiResult(json.candidates?.[0]?.content?.parts?.[0]?.text || "Erro na profecia.");
                } catch(e) { setAiResult("Erro no ritual: " + e.message); }
                setAiLoading(false);
            };

            // Helpers
            const grouped = data.reduce((acc, i) => { (acc[i.category||'GERAL'] = acc[i.category||'GERAL'] || []).push(i); return acc; }, {});

            if (loading) return <div className="h-screen flex items-center justify-center bg-slate-50"><div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div></div>;

            return (
                <div className="flex flex-col h-screen fade-in">
                    {/* Header */}
                    <div className="bg-slate-900 text-white p-4 shadow-lg sticky top-0 z-30">
                        <div className="max-w-7xl mx-auto flex flex-col sm:flex-row justify-between items-center gap-4">
                            <div className="flex items-center gap-3">
                                <div className="p-2 bg-gradient-to-br from-blue-600 to-indigo-700 rounded-lg"><Database size={20}/></div>
                                <div><h1 className="text-xl font-bold flex items-center gap-2">Estoque Ventura <span className="text-[10px] bg-orange-500 px-2 py-0.5 rounded-full">{useLocalStorage ? 'LOCAL' : 'NUVEM'}</span></h1></div>
                            </div>
                            <div className="flex gap-2 flex-wrap justify-center">
                                <button onClick={() => setShowConfig(!showConfig)} className="p-2 hover:bg-white/10 rounded"><Settings size={20}/></button>
                                <div className="w-px h-8 bg-slate-700 mx-1"></div>
                                <button onClick={handleAI} className="btn-header bg-indigo-600 hover:bg-indigo-500 text-white px-3 py-2 rounded flex gap-2"><Sparkles size={16}/> IA</button>
                                <button onClick={() => setShowAddModal(true)} className="btn-header bg-blue-600 hover:bg-blue-500 text-white px-3 py-2 rounded flex gap-2"><Plus size={16}/> Novo</button>
                                <button onClick={handleExport} className="btn-header bg-slate-700 hover:bg-slate-600 text-white px-3 py-2 rounded flex gap-2"><FileSpreadsheet size={16}/> Excel</button>
                            </div>
                        </div>
                        {showConfig && (
                            <div className="absolute top-full right-4 mt-2 w-80 bg-white text-slate-800 p-4 rounded shadow-xl border z-50 animate-in fade-in">
                                <h3 className="font-bold mb-2 flex justify-between">Configura√ß√µes <button onClick={()=>setShowConfig(false)}><X size={16}/></button></h3>
                                <div className="space-y-3">
                                    <input type="password" value={apiKey} onChange={e=>setApiKey(e.target.value)} placeholder="API Key do Gemini..." className="w-full p-2 border rounded text-sm"/>
                                    <button onClick={()=>{if(confirm('Zerar tudo?')) { saveLocal(defaultData); setShowConfig(false); }}} className="w-full bg-red-100 text-red-600 p-2 rounded font-bold text-xs flex justify-center gap-2"><RefreshCw size={14}/> RESETAR</button>
                                </div>
                            </div>
                        )}
                    </div>

                    {/* Content */}
                    <div className="flex-1 overflow-auto p-2 sm:p-4 bg-slate-100">
                        <div className="max-w-7xl mx-auto space-y-6 pb-20">
                            {Object.entries(grouped).sort().map(([cat, items]) => (
                                <div key={cat} className="bg-white shadow-sm rounded-lg border overflow-hidden">
                                    <div className="bg-slate-50 px-4 py-2 border-b flex justify-between"><span className="font-bold text-slate-700 text-sm">{cat}</span><span className="text-xs bg-white px-2 rounded border">{items.length}</span></div>
                                    <div className="overflow-x-auto">
                                        <table className="w-full text-left min-w-[800px] text-xs sm:text-sm">
                                            <thead>
                                                <tr className="bg-white border-b text-[10px] uppercase font-bold text-slate-500">
                                                    <th className="p-3">Produto</th>
                                                    <th className="p-2 text-center w-16 bg-blue-50/30">Ini</th>
                                                    <th className="p-2 text-center w-16 text-emerald-600">Ent</th>
                                                    <th className="p-2 text-center w-16 text-red-600">Vend</th>
                                                    <th className="p-2 text-center w-16 text-orange-600">Cons</th>
                                                    <th className="p-2 text-center w-16 text-orange-600">Vale</th>
                                                    <th className="p-2 text-center w-16 text-red-700">Avar</th>
                                                    <th className="p-2 text-center w-20 bg-slate-100 border-x">Calc</th>
                                                    <th className="p-2 text-center w-20 bg-yellow-50 border-r text-blue-900">Real</th>
                                                    <th className="p-2 text-center w-24">Status</th>
                                                    <th className="p-2 w-10"></th>
                                                </tr>
                                            </thead>
                                            <tbody className="divide-y divide-slate-100">
                                                {items.map(i => {
                                                    const calc = (Number(i.initial)||0)+(Number(i.entry)||0)-(Number(i.sales)||0)-(Number(i.internal)||0)-(Number(i.voucher)||0)-(Number(i.damage)||0);
                                                    const diff = calc - (Number(i.physicalCount)||0);
                                                    return (
                                                        <tr key={i.id} className="hover:bg-slate-50 group">
                                                            <td className="p-1"><input value={i.product} onChange={e=>handleUpdate(i.id,'product',e.target.value)} className="w-full bg-transparent px-2 py-1 font-semibold outline-none focus:bg-white rounded"/></td>
                                                            {['initial','entry','sales','internal','voucher','damage'].map(f=><td key={f} className="p-1"><input type="number" value={i[f]} onChange={e=>handleUpdate(i.id,f,Number(e.target.value))} className={`w-full text-center bg-transparent outline-none focus:bg-white focus:ring-1 focus:ring-blue-200 rounded ${f==='entry'?'text-emerald-700':''}`}/></td>)}
                                                            <td className="text-center font-bold bg-slate-100 border-x text-slate-600">{calc}</td>
                                                            <td className="p-1 bg-yellow-50 border-r"><input type="number" value={i.physicalCount} onChange={e=>handleUpdate(i.id,'physicalCount',Number(e.target.value))} className="w-full text-center bg-transparent font-bold text-blue-900 outline-none focus:bg-white"/></td>
                                                            <td className="text-center p-2">{diff===0?<span className="px-2 py-0.5 rounded text-[10px] font-bold bg-emerald-100 text-emerald-700">OK</span>:diff>0?<span className="px-2 py-0.5 rounded text-[10px] font-bold bg-blue-100 text-blue-700">+{diff}</span>:<span className="px-2 py-0.5 rounded text-[10px] font-bold bg-red-100 text-red-700">-{Math.abs(diff)}</span>}</td>
                                                            <td className="text-center"><button onClick={()=>handleDelete(i.id)} className="text-slate-300 hover:text-red-500"><Trash2 size={14}/></button></td>
                                                        </tr>
                                                    )
                                                })}
                                            </tbody>
                                        </table>
                                    </div>
                                </div>
                            ))}
                        </div>
                    </div>
                    
                    {/* Add Modal */}
                    {showAddModal && (
                        <div className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center p-4 backdrop-blur-sm">
                            <div className="bg-white rounded shadow-2xl w-full max-w-sm p-6 space-y-4">
                                <h3 className="font-bold text-lg flex justify-between">Novo Item <button onClick={()=>setShowAddModal(false)}><X/></button></h3>
                                <input autoFocus value={newProductName} onChange={e=>setNewProductName(e.target.value)} placeholder="NOME DO PRODUTO" className="w-full p-2 border rounded uppercase font-bold"/>
                                <select value={newCategory} onChange={e=>{setNewCategory(e.target.value);setCustomCategory('')}} className="w-full p-2 border rounded bg-slate-50"><option value="">-- Categoria --</option>{[...new Set(data.map(i=>i.category))].sort().map(c=><option key={c} value={c}>{c}</option>)}</select>
                                <input value={customCategory} onChange={e=>{setCustomCategory(e.target.value);setNewCategory('')}} placeholder="OU NOVA CATEGORIA" className="w-full p-2 border rounded uppercase text-xs"/>
                                <button onClick={handleAddRow} disabled={!newProductName} className="w-full bg-blue-600 text-white p-2 rounded font-bold hover:bg-blue-700 disabled:opacity-50">SALVAR</button>
                            </div>
                        </div>
                    )}
                    
                    {/* Import Modal */}
                    {showImportModal && (
                        <div className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center p-4 backdrop-blur-sm">
                            <div className="bg-white rounded shadow-2xl w-full max-w-sm p-6 space-y-4">
                                <h3 className="font-bold text-lg flex justify-between"><span className="flex items-center gap-2"><CloudDownload size={18}/> Sincronizar</span> <button onClick={()=>setShowImportModal(false)}><X/></button></h3>
                                <input value={importUrl} onChange={e=>setImportUrl(e.target.value)} placeholder="Link do CSV do Google Sheets..." className="w-full p-2 border rounded text-xs"/>
                                <button onClick={()=>{ alert('Importar!'); setShowImportModal(false); }} className="w-full bg-emerald-600 text-white p-2 rounded font-bold hover:bg-emerald-700">SINCRONIZAR</button>
                            </div>
                        </div>
                    )}

                    {/* AI Modal */}
                    {showAIModal && (
                        <div className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center p-4 backdrop-blur-sm">
                            <div className="bg-white rounded shadow-2xl w-full max-w-lg p-6 max-h-[80vh] overflow-auto">
                                <h3 className="font-bold text-lg flex justify-between text-indigo-700 mb-4 gap-2"><Sparkles/> Or√°culo do Mestre <button onClick={()=>setShowAIModal(false)} className="text-black"><X/></button></h3>
                                {aiLoading ? <div className="text-center py-8"><RefreshCw className="animate-spin mx-auto text-indigo-500"/></div> : <div className="whitespace-pre-wrap text-sm">{aiResult}</div>}
                            </div>
                        </div>
                    )}
                </div>
            );
        }

        const root = createRoot(document.getElementById('root'));
        root.render(<StockSheet />);
    </script>
</body>
